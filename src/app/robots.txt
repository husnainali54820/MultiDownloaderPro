import { MetadataRoute } from 'next';

export default function robots(): MetadataRoute.Robots {
  return {
    rules: {
      userAgent: '*',
      allow: '/',
      // Only block the API. Let Google see your JS/CSS so it can verify the site.
      disallow: ['/api/'], 
    },
    sitemap: 'https://mdownloader.vercel.app/sitemap.xml',
  };
}
